Settings & Sync
Supabase connection and data management
Set up Supabase — free, ~10 minutes
1
Go to supabase.com → create free account → New Project
2
In SQL Editor, run the query below
CREATE TABLE expenses ( id BIGSERIAL PRIMARY KEY, date DATE NOT NULL, month TEXT, day TEXT, item TEXT DEFAULT '', category TEXT DEFAULT '', overall_cat TEXT NOT NULL, outlet TEXT NOT NULL, paid_price NUMERIC(10,2) NOT NULL, actual_price NUMERIC(10,2), payment TEXT DEFAULT 'Card', created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE expenses ENABLE ROW LEVEL SECURITY; CREATE POLICY "Allow all" ON expenses FOR ALL TO anon, authenticated USING (true) WITH CHECK (true); GRANT ALL ON expenses TO anon; GRANT USAGE ON SEQUENCE expenses_id_seq TO anon; CREATE TABLE outlets (id BIGSERIAL PRIMARY KEY, name TEXT UNIQUE NOT NULL); ALTER TABLE outlets ENABLE ROW LEVEL SECURITY; CREATE POLICY "Allow all" ON outlets FOR ALL TO anon, authenticated USING (true) WITH CHECK (true); GRANT ALL ON outlets TO anon; GRANT USAGE ON SEQUENCE outlets_id_seq TO anon;
3
Go to Project Settings → API → copy Project URL and anon key below
